import nltk
import random
import re
nltk.download('punkt')
# Sample data & patterns
greetings = ['hi', 'hello', 'hey', 'good morning', 'good evening']
greeting_responses = ['Hello! How can I help you with your finances today?',
                      'Hi there! Need help with savings, taxes, or investments?',
                      'Hey! What can I do for your financial planning?']
# Tax advice
def tax_advice(income):
    if income < 10000:
        return "You're likely in a low tax bracket. Consider a Roth IRA for tax-free growth."
    elif income < 40000:
        return "You might benefit from tax deductions like student loan interest or retirement contributions."
    else:
        return "Consider talking to a tax professional to optimize deductions and credits."
# Savings advice
def savings_advice(amount):
    if amount < 1000:
        return "Start by building an emergency fund with at least $1,000."
    elif amount < 10000:
        return "Good! Try to build 3-6 months of expenses as emergency savings."
    else:
        return "You're doing well. Consider high-yield savings or short-term investments."
# Investment advice
def investment_advice(age):
    if age < 30:
        return "You can afford higher risk. Consider index funds or stocks."
    elif age < 50:
        return "Balance risk with stability. Look at bonds and mutual funds."
    else:
        return "Preserve capital. Focus on low-risk investments like bonds or dividend stocks."
# Simple keyword extraction and response
def get_response(user_input):
    user_input = user_input.lower()
    if any(greet in user_input for greet in greetings):
        return random.choice(greeting_responses)
    if 'tax' in user_input or 'deduction' in user_input:
        income_match = re.search(r'\d+', user_input)
        income = int(income_match.group()) if income_match else 30000  # Default income
        return tax_advice(income)
    if 'save' in user_input or 'savings' in user_input:
        amount_match = re.search(r'\d+', user_input)
        amount = int(amount_match.group()) if amount_match else 500  # Default savings
        return savings_advice(amount)
    if 'invest' in user_input or 'investment' in user_input:
        age_match = re.search(r'\d+', user_input)
        age = int(age_match.group()) if age_match else 35  # Default age
        return investment_advice(age)
    return "Sorry, I didn’t quite get that. Ask me about savings, taxes, or investments!"
# Chat loop
def chat():
    print("👋 Welcome to your Personal Finance Chatbot!")
    print("Ask me anything about savings, taxes, or investments. Type 'exit' to quit.")
    while True:
        user_input = input("You: ")
        if user_input.lower() in ['exit', 'quit']:
            print("Chatbot: Take care of your money. Goodbye!")
            break
        response = get_response(user_input)
        print("Chatbot:", response)
if __name__ == "__main__":
    chat()
